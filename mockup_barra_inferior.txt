<script>

let pagesRead = 12;
const totalPages = 560;

function changePages(value){

    pagesRead += value;

    if(pagesRead < 0){
        pagesRead = 0;
    }

    if(pagesRead > totalPages){
        pagesRead = totalPages;
    }

    document.getElementById("currentPages").textContent = pagesRead;

    document.getElementById("pageProgress").textContent =
        `${pagesRead} / ${totalPages} páginas`;

    let percent = (pagesRead / totalPages) * 100;

    document.getElementById("progressBar").style.width =
        percent + "%";
}

/* Cronómetro */

let seconds = 1500; // 25 minutos
let interval;

function formatTime(){

    let mins = Math.floor(seconds / 60);
    let secs = seconds % 60;

    mins = String(mins).padStart(2,'0');
    secs = String(secs).padStart(2,'0');

    document.getElementById("timer").textContent =
        `${mins}:${secs}`;
}

function startTimer(){

    clearInterval(interval);

    interval = setInterval(() => {

        seconds++;

        let mins = Math.floor(seconds / 60);
        let secs = seconds % 60;

        mins = String(mins).padStart(2,'0');
        secs = String(secs).padStart(2,'0');

        document.getElementById("timer").textContent =
            `${mins}:${secs}`;

    },1000);
}

function pauseTimer(){
    clearInterval(interval);
}

function stopTimer(){

    clearInterval(interval);

    alert(
        `Sesión finalizada\n\nTiempo: ${document.getElementById("timer").textContent}\nPáginas leídas: ${pagesRead}`
    );
}

document.getElementById("playBtn").addEventListener("click", startTimer);
document.getElementById("pauseBtn").addEventListener("click", pauseTimer);
document.getElementById("stopBtn").addEventListener("click", stopTimer);

formatTime();

</script>
